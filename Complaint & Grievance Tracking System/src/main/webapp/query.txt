CREATE TABLE COMPLAINT_TB (
    COMPLAINTID   VARCHAR2(12 BYTE) PRIMARY KEY,
    COMPLAINANT   VARCHAR2(50 BYTE) NOT NULL,
    TITLE         VARCHAR2(80 BYTE) NOT NULL,
    DEPARTMENT    VARCHAR2(40 BYTE) NOT NULL,
    COMPLAINTDATE DATE NOT NULL,
    STATUS        VARCHAR2(20 BYTE) NOT NULL,
    REMARKS       VARCHAR2(200 BYTE)
);
CREATE SEQUENCE COMPLAINT_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;
